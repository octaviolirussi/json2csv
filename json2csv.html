json2csv

json2csv – JSON to CSV converter.

This package exposes the public API for converting JSON data to CSV format. Use convert_records(), convert_string(), or convert_file() depending on whether your data is already in memory or stored on disk.

Example::

from json2csv import convert_records
csv_text = convert_records([{"name": "Ana", "age": 30}])
print(csv_text)

158def convert_records(
159    records: list[dict[str, Any]],
160    *,
161    flatten: bool = False,
162    sep: str = ".",
163    dialect: str = "excel",
164    extrasaction: Literal["raise", "ignore"] = "ignore",
165) -> str:
166    """Convert a list of dicts to a CSV string.
167
168    Args:
169        records: Non-empty list of ``dict`` objects representing rows.
170        flatten: If ``True``, nested dicts/lists are flattened using *sep*.
171        sep: Separator used when generating dotted keys for nested fields.
172        dialect: ``csv`` module dialect name (default ``"excel"``).
173        extrasaction: Passed to :class:`csv.DictWriter`; ``"ignore"`` silently
174            drops keys not in the header row.
175
176    Returns:
177        A string containing the full CSV output including the header row.
178
179    Raises:
180        InvalidJsonStructureError: If any element of *records* is not a dict.
181        EmptyDataError: If *records* is empty.
182
183    Examples:
184        >>> result = convert_records([{"name": "Ana", "age": 30}])
185        >>> "name" in result and "Ana" in result
186        True
187    """
188    validated = _validate_records(records)
189    rows: list[dict[str, Any]] = (
190        [_flatten(r, sep=sep) for r in validated] if flatten else validated
191    )
192    fieldnames = _build_fieldnames(validated, flatten=flatten, sep=sep)
193
194    buf = io.StringIO()
195    writer = csv.DictWriter(
196        buf,
197        fieldnames=fieldnames,
198        dialect=dialect,
199        extrasaction=extrasaction,
200    )
201    writer.writeheader()
202    writer.writerows(rows)
203    return buf.getvalue()

def convert_file( source: str | pathlib._local.Path, destination: str | pathlib._local.Path, *, flatten: bool = False, sep: str = '.', dialect: str = 'excel', encoding: str = 'utf-8') -> int: 
233def convert_file(
234    source: str | Path,
235    destination: str | Path,
236    *,
237    flatten: bool = False,
238    sep: str = ".",
239    dialect: str = "excel",
240    encoding: str = "utf-8",
241) -> int:
242    """Read a JSON file and write the result to a CSV file.
243
244    Args:
245        source: Path to the input ``.json`` file.
246        destination: Path for the output ``.csv`` file.  Parent directories
247            are created automatically when they do not exist.
248        flatten: Flatten nested structures when ``True``.
249        sep: Key separator used during flattening.
250        dialect: CSV dialect name.
251        encoding: Character encoding for reading *source* and writing
252            *destination* (default ``"utf-8"``).
253
254    Returns:
255        Number of data rows written (excluding the header).
256
257    Raises:
258        FileReadError: If *source* does not exist, cannot be read, or contains
259            invalid JSON.
260        FileWriteError: If *destination* cannot be written.
261        InvalidJsonStructureError: If the JSON root is not an array of objects.
262        EmptyDataError: If the JSON array is empty.
263    """
264    src = Path(source)
265    dst = Path(destination)
266
267    logger.info("Reading source: %s", src)
268    try:
269        raw = src.read_text(encoding=encoding)
270    except OSError as exc:
271        raise FileReadError(f"Cannot read {src!r}: {exc}") from exc
272
273    csv_text = convert_string(raw, flatten=flatten, sep=sep, dialect=dialect)
274
275    logger.info("Writing destination: %s", dst)
276    try:
277        dst.parent.mkdir(parents=True, exist_ok=True)
278        dst.write_text(csv_text, encoding=encoding)
279    except OSError as exc:
280        raise FileWriteError(f"Cannot write {dst!r}: {exc}") from exc
281
282    row_count = csv_text.count("\n") - 1  # subtract header line
283    logger.info("Wrote %d data rows to %s", row_count, dst)
284    return max(row_count, 0)
